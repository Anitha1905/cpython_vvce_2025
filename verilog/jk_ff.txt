module JKFF (
  input clk,
  input rst_n,
  input J,
  input K,
  output reg Q,
  output Q_bar
);
  always @(posedge clk or negedge rst_n) begin
    if (!rst_n) begin
      Q <= 0;
    end else begin
      case ({J, K})
        2'b00: Q <= Q;       
        2'b01: Q <= 0;      
        2'b10: Q <= 1;       
        2'b11: Q <= ~Q;     
        default: Q <= Q;
      endcase
    end
  end
  assign Q_bar = ~Q;
endmodule
